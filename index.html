backgroundLoader

const queue = [];
let isWaiting = false;
const requestLoad = () => {
    if (isWaiting || !queue.length) { return; }
    const loader = queue.pop();
    isWaiting = true;
    setTimeout(() => {
        loader(() => {}, () => {
            isWaiting = false;
            requestLoad();
        });
    }, 200);
};
export default loader => {
    queue.push(loader);
    requestLoad();
};
